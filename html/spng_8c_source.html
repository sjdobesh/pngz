225 int spng_load(spng* s, char* path) {

228  png_structp png;
229  png_infop info;
230  unsigned width, height;
231  unsigned char bit_depth, color_type;
232  unsigned char** byte_ptrs;
233  errno = 0;
234  fprintf(stderr, "spng loading png at '%s'\n", path);
235 
236  /* open the file and read into info struct */
237  if(!(fp = fopen(path, "rb"))) {
238  fprintf(stderr, "ERROR > opening file.\n");
239  errno = EIO;
240  return 1;
241  }
242  if (!(png = png_create_read_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL))) {
243  fprintf(stderr, "ERROR > creating read struct.\n");
244  errno = EIO;
245  return 1;
246  }
247  if (!(info = png_create_info_struct(png))) {
248  fprintf(stderr, "ERROR > creating png info.\n");
249  errno = EIO;
250  return 1;
251  }
252  if (setjmp(png_jmpbuf(png))) {
253  fprintf(stderr, "ERROR > io failed.\n");
254  errno = EIO;
255  return 1;
256  }
257 
258  /* read in png info */
259  png_init_io(png, fp);
260  png_read_info(png, info);
261  width = png_get_image_width(png, info);
262  height = png_get_image_height(png, info);
263  bit_depth = png_get_bit_depth(png, info);
264  color_type = png_get_color_type(png, info);
265 
266  /* convert contents into RGBA8 */
267  if (color_type == PNG_COLOR_TYPE_PALETTE) {
268  png_set_palette_to_rgb(png);
269  }
270  if (bit_depth == 16) {
271  png_set_strip_16(png);
272  }
273  if (bit_depth < 8) {
274  if (color_type == PNG_COLOR_TYPE_GRAY) {
275  png_set_expand_gray_1_2_4_to_8(png);
276  } else {
277  png_set_expand(png);
278  }
279  }
280  if (png_get_valid(png, info, PNG_INFO_tRNS)){
281  png_set_tRNS_to_alpha(png);
282  }
283  if (
284  color_type != PNG_COLOR_TYPE_RGBA ||
285  color_type != PNG_COLOR_TYPE_GRAY_ALPHA
286  ) {
287  png_set_filler(png, 0xFF, PNG_FILLER_AFTER);
288  }
289  if (
290  color_type != PNG_COLOR_TYPE_GRAY ||
291  color_type != PNG_COLOR_TYPE_GRAY_ALPHA
292  ) {
293  png_set_gray_to_rgb(png);
294  }
295  png_read_update_info(png, info);
296 
297  /* malloc pixel buffers */
298  byte_ptrs = spng_alloc_bytes(height, width * 4);
299  s->pixels = spng_alloc_pixels(height, width);
300  /* load into byte buffer, transfer to pixels */
301  png_read_image(png, byte_ptrs);
302  spng_pack_pixels(byte_ptrs, s->pixels, height, width);
303  /* clean up */
304  spng_free_bytes(byte_ptrs, height);
305  png_destroy_read_struct(&png, &info, NULL);
306  fclose(fp);
307 
308  /* if we got here, pack referenced struct and return */
309  s->width = width;
310  s->height = height;
311  return 0;
312 }
